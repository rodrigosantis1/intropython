#dicionarios com os estilos das bandas, se ha + de 1, entao eh misto
misto = {}
puro = {}
todos = {}
for tipo in bandas['style']:
    subs = tipo.split(',') 
    for ind in subs:
            todos[ind] = todos.get(ind, 0) + 1
    if (len(subs) == 1):
        puro[subs[0]] = puro.get(subs[0], 0) + 1
    else:
        for ind in subs:
            misto[ind] = misto.get(ind, 0) + 1
tipos_df = pd.DataFrame()
tipos_df['mistos'] = pd.Series(misto)
tipos_df['puros'] = pd.Series(puro)
tipos_df['todos'] = pd.Series(todos)
tipos_df = tipos_df.sort_values(by='mistos', ascending=False)[:10]
fig = plt.figure()
st = fig.suptitle("Maiores generos de bandas", fontsize="x-large")
